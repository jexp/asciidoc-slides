++++

[options="step"]
Cypher

[source,options="step"]
----
MATCH (a:NaturalPerson)-[:MEMBER]->(u:OrganizationalUnit),
  u<-[:MEMBER]-(b:NaturalPerson)
WHERE a.canonicalName = "Alice Miller"
RETURN b.canonicalName;
----
