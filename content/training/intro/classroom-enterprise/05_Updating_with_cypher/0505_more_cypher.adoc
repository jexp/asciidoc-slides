== Create label-specific index

[source,cypher,options="step"]
----
CREATE INDEX ON :Human(name);
----

== Shortest path

[source,cypher,options="step"]
----
MATCH (h1:Human {name: "Alice"}), 
  (h2:Human {name: "Charlene"}),  
  p=shortestPath(h1-[:EMPLOYEE|LEAD*]-h2)
RETURN length(nodes(p));
----
